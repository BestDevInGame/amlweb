function DepositScreen({ smartAccountAddress, ethBalance, tokens }) {
  return (
    <div>
      <h2>Deposit Funds</h2>
      <p>Send your assets to your personal vault to get started</p>

      {/* Show the smart account as "their vault" */}
      <div className="vault-address">
        <p>Your Vault Address</p>
        <code>{smartAccountAddress}</code>
      </div>

      {/* ETH deposit */}
      <div className="deposit-item">
        <span>ETH Balance: {formatEther(ethBalance)}</span>
        <button onClick={() => depositETH(ethBalance)}>
          Deposit ETH
        </button>
      </div>

      {/* Token deposits */}
      {tokens.map(token => (
        <div key={token.address} className="deposit-item">
          <span>{token.symbol} Balance: {token.balance}</span>
          <button onClick={() => depositToken(token)}>
            Deposit {token.symbol}
          </button>
        </div>
      ))}

      <small>
        These funds are stored in your personal smart account.
        Only you can authorize withdrawals.
      </small>
    </div>
  )
}
